```mermaid
graph TD
    %% 定义样式
    classDef layerBox fill:#f9f9f9,stroke:#333,stroke-width:2px;
    classDef highlight fill:#e1f5fe,stroke:#01579b,stroke-width:2px;
    classDef base fill:#fff3e0,stroke:#e65100,stroke-width:2px;

    %% 第一层：驾驶数据与分类层
    subgraph Layer1 [驾驶数据与分类层]
        D1[Kangle 平台原始数据] --> D2[特征提取: 车速/加速度/踏板等]
        D2 --> D3[机器学习分类器: Eco/Normal/Sport]
    end

    %% 第二层：场景与驾驶员模型层
    subgraph Layer2 [场景与驾驶员模型层]
        S1[随机道路生成: 坡度/弯道/限速] --> S2[参考驾驶员模型: 规则驱动]
        S2 --> S3[输出参考扭矩 T_ref]
    end

    %% 第三层：核心控制算法层
    subgraph Layer3 [核心控制算法层]
        A1[29维状态观测 S_k] --> A2[PPO 强化学习智能体]
        A2 --> A3[输出残差扭矩 ΔT_k]
        A3 --> A4[奖励评估: 跟踪惩罚 + 节能奖励]
    end

    %% 第四层：物理与动力学环境层 (基座)
    subgraph Layer4 [物理与动力学环境层 - 本章重点]
        P1[纵向动力学受力分析] --- P2[电机效率地图 η-Map]
        P2 --- P3[SOC 与能耗实时更新]
    end

    %% 连线关系
    Layer1 -->|风格标签| Layer2
    Layer2 -->|T_ref| Layer3
    Layer3 -->|指令扭矩 T_cmd = T_ref + ΔT_k| Layer4
    Layer4 -.->|状态反馈 S_k+1 / 奖励 r_k| Layer3

    %% 应用样式
    class Layer1,Layer2 layerBox;
    class Layer3 highlight;
    class Layer4 base;
```
